115 LogicalResult LLVMFunctionType::verifyConstructionInvariants(
116  Location loc, LLVMType result, ArrayRef<LLVMType> arguments, bool) {
117  if (!isValidResultType(result))
118  return emitError(loc, "invalid function result type: ") << result;
119 
120  for (LLVMType arg : arguments)
121  if (!isValidArgumentType(arg))
122  return emitError(loc, "invalid function argument type: ") << arg;
123 
124  return success();

199 LLVMStructType LLVMStructType::getNewIdentified(MLIRContext *context,
200  StringRef name,
201  ArrayRef<LLVMType> elements,
202  bool isPacked) {
203  std::string stringName = name.str();
204  unsigned counter = 0;
205  do {
206  auto type = LLVMStructType::getIdentified(context, stringName);
207  if (type.isInitialized() || failed(type.setBody(elements, isPacked))) {
208  counter += 1;
209  stringName = (Twine(name) + "." + std::to_string(counter)).str();
210  continue;
211  }
212  return type;
213  } while (true);
214 }

348 llvm::TypeSize mlir::LLVM::getPrimitiveTypeSizeInBits(Type type) {
349  assert(isCompatibleType(type) &&
350  "expected a type compatible with the LLVM dialect");
351 
352  return llvm::TypeSwitch<Type, llvm::TypeSize>(type)
353  .Case<LLVMHalfType, LLVMBFloatType>(
354  [](LLVMType) { return llvm::TypeSize::Fixed(16); })
355  .Case<LLVMFloatType>([](LLVMType) { return llvm::TypeSize::Fixed(32); })
356  .Case<LLVMDoubleType, LLVMX86MMXType>(
357  [](LLVMType) { return llvm::TypeSize::Fixed(64); })
358  .Case<LLVMIntegerType>([](LLVMIntegerType intTy) {
359  return llvm::TypeSize::Fixed(intTy.getBitWidth());
360  })
361  .Case<LLVMX86FP80Type>([](LLVMType) { return llvm::TypeSize::Fixed(80); })
362  .Case<LLVMPPCFP128Type, LLVMFP128Type>(
363  [](LLVMType) { return llvm::TypeSize::Fixed(128); })
364  .Case<LLVMVectorType>([](LLVMVectorType t) {
365  llvm::TypeSize elementSize =
366  getPrimitiveTypeSizeInBits(t.getElementType());
367  llvm::ElementCount elementCount = t.getElementCount();
368  assert(!elementSize.isScalable() &&
369  "vector type should have fixed-width elements");
370  return llvm::TypeSize(elementSize.getFixedSize() *
371  elementCount.getKnownMinValue(),
372  elementCount.isScalable());
373  })
374  .Default([](Type ty) {
375  assert((ty.isa<LLVMVoidType, LLVMLabelType, LLVMMetadataType,
376  LLVMTokenType, LLVMStructType, LLVMArrayType,
377  LLVMPointerType, LLVMFunctionType>()) &&
378  "unexpected missing support for primitive type");
379  return llvm::TypeSize::Fixed(0);
380  });
381 }
